<script>
let queryResult = null;

function loadSample() {
    fetch('/trace/api/uniswap/sample')
        .then(response => response.json())
        .then(data => {
            document.getElementById('address-input').value = data.sample_address;
        });
}

function traceSwaps() {
    const address = document.getElementById('address-input').value.trim();
    if (!address) {
        alert('请输入ETH地址');
        return;
    }
    
    document.getElementById('loading').classList.remove('hidden');
    document.getElementById('error-display').classList.add('hidden');
    document.getElementById('results').classList.add('hidden');
    
    fetch('/trace/api/uniswap/query', {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({address: address})
    })
    .then(response => response.json())
    .then(data => {
        document.getElementById('loading').classList.add('hidden');
        if (data.success) {
            queryResult = data;
            displayResults(data);
        } else {
            showError(data.error || '追踪失败');
        }
    })
    .catch(error => {
        document.getElementById('loading').classList.add('hidden');
        showError('请求失败: ' + error.message);
    });
}

function showError(message) {
    document.getElementById('error-message').textContent = message;
    document.getElementById('error-display').classList.remove('hidden');
}

function displayResults(data) {
    document.getElementById('results').classList.remove('hidden');
    
    // Display basic info
    document.getElementById('basic-info').innerHTML = `
        <div class="bg-gray-50 p-3 rounded">
            <span class="text-sm text-gray-500">追踪地址</span>
            <p class="font-bold text-gray-800 text-sm">${data.address}</p>
        </div>
        <div class="bg-gray-50 p-3 rounded">
            <span class="text-sm text-gray-500">Swap交易数</span>
            <p class="font-bold text-gray-800">${(data.swaps || []).length}</p>
        </div>
    `;
    
    // Display swap cards
    displaySwapCards(data.swaps || []);
    
    // Display flow diagram
    document.getElementById('diagram-content').textContent = data.flow_diagram || '无数据';
}

function displaySwapCards(swaps) {
    const container = document.getElementById('swaps-container');
    container.innerHTML = swaps.map(swap => `
        <div class="bg-gray-50 rounded p-4 border border-gray-200">
            <div class="flex justify-between items-start mb-2">
                <p class="font-bold text-gray-800">${swap.type || 'Swap交易'}</p>
                <span class="text-xs text-gray-500">${swap.time || '未知'}</span>
            </div>
            <div class="grid grid-cols-2 gap-2 text-sm">
                <div><span class="text-gray-500">交易哈希:</span> <span class="text-xs">${swap.hash || '未知'}</span></div>
                <div><span class="text-gray-500">输入:</span> <span class="font-medium">${swap.amount_in || '未知'}</span></div>
                <div><span class="text-gray-500">输出:</span> <span class="font-medium">${swap.amount_out || '未知'}</span></div>
                <div><span class="text-gray-500">Token:</span> <span>${swap.from_token || '?'} → ${swap.to_token || '?'}</span></div>
            </div>
        </div>
    `).join('');
}

function exportJSON() {
    if (!queryResult) return alert('请先进行追踪');
    fetch('/trace/api/uniswap/export/json', {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({result: queryResult})
    })
    .then(response => response.blob())
    .then(blob => {
        const url = window.URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = 'uniswap_trace.json';
        a.click();
    });
}

function exportCSV() {
    if (!queryResult) return alert('请先进行追踪');
    fetch('/trace/api/uniswap/export/csv', {
        method: 'POST',
        headers: {'Content-Type': 'application/json'},
        body: JSON.stringify({result: queryResult})
    })
    .then(response => response.blob())
    .then(blob => {
        const url = window.URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = 'uniswap_trace.csv';
        a.click();
    });
}

                    ${out.wallet_hint ? `<span class="text-xs text-gray-500 ml-2">${out.wallet_hint}</span>` : ''}
                </div>
                <div class="text-right">
                    <span class="font-bold">${valueBTC} BTC</span>
