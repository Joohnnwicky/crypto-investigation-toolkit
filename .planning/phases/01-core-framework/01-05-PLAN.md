<script>
let analysisResult = null;

function loadSample() {
    fetch('/tron/api/sample')
        .then(response => response.json())
        .then(data => {
            document.getElementById('address-input').value = data.address;
        })
        .catch(error => {
            alert('加载样本失败: ' + error.message);
        });
}

function analyzeAddress() {
    const address = document.getElementById('address-input').value.trim();
    
    if (!address) {
        alert('请输入TRON地址');
        return;
    }
    
    // Show loading
    document.getElementById('loading').classList.remove('hidden');
    document.getElementById('error-display').classList.add('hidden');
    document.getElementById('results').classList.add('hidden');
    
    fetch('/tron/api/analyze', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json'
        },
        body: JSON.stringify({ address: address })
    })
    .then(response => response.json())
    .then(data => {
        document.getElementById('loading').classList.add('hidden');
        
        if (data.success) {
            analysisResult = data;
            displayResults(data);
        } else {
            showError(data.error || '分析失败');
        }
    })
    .catch(error => {
        document.getElementById('loading').classList.add('hidden');
        showError('请求失败: ' + error.message);
    });
}

function showError(message) {
    document.getElementById('error-message').textContent = message;
    document.getElementById('error-display').classList.remove('hidden');
}

function displayResults(data) {
    document.getElementById('results').classList.remove('hidden');
    
    // Display basic info
    const basicInfo = document.getElementById('basic-info');
    basicInfo.innerHTML = `
        <div class="bg-gray-50 p-3 rounded">
            <span class="text-sm text-gray-500">TRX余额</span>
            <p class="font-bold text-gray-800">${data.basic_info.balance.toFixed(6)} TRX</p>
        </div>
        <div class="bg-gray-50 p-3 rounded">
            <span class="text-sm text-gray-500">USDT余额</span>
            <p class="font-bold text-gray-800">${data.basic_info.usdt_balance.toFixed(2)} USDT</p>
        </div>
        <div class="bg-gray-50 p-3 rounded">
            <span class="text-sm text-gray-500">总交易数</span>
            <p class="font-bold text-gray-800">${data.basic_info.total_transaction_count}</p>
        </div>
        <div class="bg-gray-50 p-3 rounded">
            <span class="text-sm text-gray-500">分析地址</span>
            <p class="font-bold text-gray-800 text-sm">${data.address}</p>
        </div>
    `;
    
    // Display score
    const score = data.alerts.score;
    const scoreCircle = document.getElementById('score-circle');
    const scoreLabel = document.getElementById('score-label');
    
    scoreCircle.textContent = score;
    
    if (score >= 70) {
        scoreCircle.className = 'w-32 h-32 rounded-full flex items-center justify-center text-3xl font-bold bg-red-500 text-white';
        scoreLabel.textContent = '高风险';
        scoreLabel.className = 'ml-4 text-xl font-medium text-red-600';
    } else if (score >= 40) {
        scoreCircle.className = 'w-32 h-32 rounded-full flex items-center justify-center text-3xl font-bold bg-yellow-500 text-white';
        scoreLabel.textContent = '中等风险';
        scoreLabel.className = 'ml-4 text-xl font-medium text-yellow-600';
    } else {
        scoreCircle.className = 'w-32 h-32 rounded-full flex items-center justify-center text-3xl font-bold bg-green-500 text-white';
        scoreLabel.textContent = '低风险';
        scoreLabel.className = 'ml-4 text-xl font-medium text-green-600';
    }
    
    // Display alerts
    displayAlerts('red', data.alerts.red || []);
    displayAlerts('yellow', data.alerts.yellow || []);
    displayAlerts('green', data.alerts.green || []);
}

function displayAlerts(type, alerts) {
    const container = document.getElementById(`${type}-alerts`);
    const list = document.getElementById(`${type}-list`);
    
    if (alerts.length === 0) {
        container.classList.add('hidden');
        return;
    }
    
    container.classList.remove('hidden');
    list.innerHTML = alerts.map(alert => `
        <div class="bg-white rounded p-4 shadow-sm">
            <p class="font-bold text-gray-800">${alert.feature}</p>
            <p class="text-sm text-gray-600 mt-1"><strong>详情:</strong> ${alert.detail}</p>
            <p class="text-sm text-gray-500 mt-1"><strong>意义:</strong> ${alert.meaning}</p>
        </div>
    `).join('');
}

function exportJSON() {
    if (!analysisResult) {
        alert('请先进行地址分析');
        return;
    }
    
    fetch('/tron/api/export/json', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json'
        },
        body: JSON.stringify({ result: analysisResult })
    })
    .then(response => {
        const filename = response.headers.get('Content-Disposition').split('filename=')[1].replace(/"/g, '');
        return response.blob().then(blob => ({ blob, filename }));
    })
    .then(({ blob, filename }) => {
        const url = window.URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = filename;
        a.click();
        window.URL.revokeObjectURL(url);
    })
    .catch(error => {
        alert('导出失败: ' + error.message);
    });
}

function exportCSV() {
    if (!analysisResult) {
        alert('请先进行地址分析');
        return;
    }
    
    fetch('/tron/api/export/csv', {
        method: 'POST',
        headers: {
            'Content-Type': 'application/json'
        },
        body: JSON.stringify({ result: analysisResult })
    })
    .then(response => {
        const filename = response.headers.get('Content-Disposition').split('filename=')[1].replace(/"/g, '');
        return response.blob().then(blob => ({ blob, filename }));
    })
    .then(({ blob, filename }) => {
        const url = window.URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
        a.download = filename;
        a.click();
        window.URL.revokeObjectURL(url);
    })
    .catch(error => {
        alert('导出失败: ' + error.message);
    });
}
